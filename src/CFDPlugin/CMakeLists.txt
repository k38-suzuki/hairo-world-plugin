set(sources
  CFDPlugin.cpp
  CFDSimulatorItem.cpp
  FlightEventReader.cpp
  Rotor.cpp
  SimplePilot.cpp
  Thruster.cpp
  WingDevice.cpp
)

set(headers
  CFDSimulatorItem.h
  FlightEventReader.h
  Rotor.h
  SimplePilot.h
  Thruster.h
  WingDevice.h

)
